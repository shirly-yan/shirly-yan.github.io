数组 基本用法和排序

objective-c

1.创建数组

[objc] view
plaincopy1. // 创建一个空的数组

  1. NSArray *array = [NSArray array];
  2. // 创建有1个元素的数组
  3. array = [NSArray arrayWithObject:@”123”];
  4. // 创建有多个元素的数组
  5. array = [NSArray arrayWithObjects:@”a”, @”b”, @”c”, nil nil];
  6. NSArray *array3 = [array arrayByAddingObjectsFromArray:[NSArray arrayWithObjects:@”4”, @”5”, nil nil]];
  8. NSArray *array4 = [NSArray arrayWithObjects:@”1”, @”2”, @”3”, @”4”, nil nil];
  9. NSRange range = NSMakeRange(1, 2);
  10. NSArray *array5 = [array4 subarrayWithRange:range];

2.数组的一些基本方法

[objc] view
plaincopy1. int count = [array count];//个数

  1. // 判断是否包含了某个元素
  2. if ([array containsObject:@”a”]) {
  3. NSLog(@”包含了字符串a”);
  4. }
  5. NSString *last = [array lastObject];最后一个元素
  6. NSString *str = [array objectAtIndex:1];根据索引获取数组中的元素
  7. int index = [array indexOfObject:@”c”];获取指定元素的索引
  8. // 让数组里面的所有对象都调用test方法，123为参数
  9. NSArray *array = [NSArray arrayWithObjects:stu1, stu2, stu3, nil nil];
  10. [array makeObjectsPerformSelector:@selector(test2:) withObject:@”123”];
  11. NSArray *array = [NSArray arrayWithObjects:@”1”, @”2”, @”3”, @”4”, nil nil];
  12. // 1-2-3-4
  13. // 利用分隔符-拼接所有的数组元素
  14. NSString *str = [array componentsJoinedByString:@”-“];
  15. // 将一个数组写入文件（生成的是一个xml文件）
  16. NSString *path = @”/Users/apple/Desktop/array.xml”;
  17. [array writeToFile:path atomically:YES];
  18. path = @”/Users/apple/Desktop/array.txt”;
  19. // 从文件中读取数组内容（文件有严格的格式要求）
  20. NSArray *array2 = [NSArray arrayWithContentsOfFile:path];

3.遍历数组

[objc] view
plaincopy1. #pragma mark 遍历数组1

  1. void arrayFor1() {
  2. NSArray *array = [NSArray arrayWithObjects:stu1, @”1”, @”2”, @”3”, nil nil];
  3. int count = array.count;
  4. for (int i = 0; i<count; i++) {
  5. id obj = [array objectAtIndex:i];
  6. }
  7. }
  9. #pragma mark 遍历数组2 快速遍历
  10. void arrayFor2() {
  11. Student *stu1 = [Student student];
  12. NSArray *array = [NSArray arrayWithObjects:stu1, @”1”, @”2”, @”3”, nil nil];
  13. int i =0;
  14. for (id obj in array) {
  15. NSLog(@”%i-%@”, i, obj);
  16. i++;
  17. }
  18. }
  20. #pragma mark 遍历数组3
  21. void arrayFor3() {
  22. Student *stu1 = [Student student];
  23. NSArray *array = [NSArray arrayWithObjects:stu1, @”1”, @”2”, @”3”, nil nil];
  24. [array enumerateObjectsUsingBlock:
  25. ^(id obj, NSUInteger idx, BOOLBOOL *stop) {
  26. NSLog(@”%i-%@”, idx, obj);
  27. // 如果索引为1，就停止遍历
  28. if (idx == 1) {
  29. // 利用指针修改外面BOOL变量的值
  30. *stop = YES;
  31. }
  32. }];
  33. }
  35. #pragma mark 遍历数组4
  36. void arrayFor4() {
  37. Student *stu1 = [Student student];
  38. NSArray *array = [NSArray arrayWithObjects:stu1, @”1”, @”2”, @”3”, nil nil];
  39. // 获取数组的迭代器
  40. // NSEnumerator *enumerator = [array objectEnumerator];
  41. // 反序迭代器（从尾部开始遍历元素）
  42. NSEnumerator *enumerator = [array reverseObjectEnumerator];
  43. // allObjects是取出没有被遍历过的对象
  44. NSArray *array2 = [enumerator allObjects];
  45. NSLog(@”array2:%@”, array2);
  46. // 获取下一个需要遍历的元素
  47. id obj = nil;
  48. while (obj = [enumerator nextObject]) {
  49. NSLog(@”obj=%@”, obj);
  50. }
  51. }

4.数组排序

[objc] view
plaincopy1. #pragma mark 数组排序1

  1. void arraySort1() {
  2. NSArray *array = [NSArray arrayWithObjects:@”2”, @”3”, @”1”, @”4”, nil nil];
  4. // 返回一个排好序的数组，原来数组的元素顺序不会改变
  5. // 指定元素的比较方法：compare:
  6. NSArray *array2 = [array sortedArrayUsingSelector:@selector(compare:)];
  7. NSLog(@”array2:%@”, array2);
  8. }
  10. #pragma mark 数组排序2
  11. void arraySort2() {
  12. Student *stu1 = [Student studentWithFirstname:@”MingJie” lastname:@”Li”];
  13. Student *stu2 = [Student studentWithFirstname:@”LongHu” lastname:@”Huang”];
  14. Student *stu3 = [Student studentWithFirstname:@”LianJie” lastname:@”Li”];
  15. Student *stu4 = [Student studentWithFirstname:@”Jian” lastname:@”Xiao”];
  16. NSArray *array = [NSArray arrayWithObjects:stu1,stu2,stu3, stu4, nil nil];
  17. // 指定排序的比较方法
  18. NSArray *array2 = [array sortedArrayUsingSelector:@selector(compareStudent:)];
  19. NSLog(@”array2:%@”, array2);
  20. }
    • (NSComparisonResult)compareStudent:(Student *)stu {
  22. // 先按照姓排序
  23. NSComparisonResult result = [self.lastname compare:stu.lastname];
  24. // 如果有相同的姓，就比较名字
  25. if (result == NSOrderedSame) {
  26. result = [self.firstname compare:stu.firstname];
  27. }
  28. return result;
  29. }
  31. #pragma mark 数组排序3
  32. void arraySort3() {
  33. Student *stu1 = [Student studentWithFirstname:@”MingJie” lastname:@”Li”];
  34. Student *stu2 = [Student studentWithFirstname:@”LongHu” lastname:@”Huang”];
  35. Student *stu3 = [Student studentWithFirstname:@”LianJie” lastname:@”Li”];
  36. Student *stu4 = [Student studentWithFirstname:@”Jian” lastname:@”Xiao”];
  37. NSArray *array = [NSArray arrayWithObjects:stu1,stu2,stu3, stu4, nil nil];
  39. // 利用block进行排序
  40. NSArray *array2 = [array sortedArrayUsingComparator:
  41. ^NSComparisonResult(Student obj1, Student obj2) {
  42. // 先按照姓排序
  43. NSComparisonResult result = [obj1.lastname compare:obj2.lastname];
  44. // 如果有相同的姓，就比较名字
  45. if (result == NSOrderedSame) {
  46. result = [obj1.firstname compare:obj2.firstname];
  47. }
  49. return result;
  50. }];
  52. NSLog(@”array2:%@”, array2);
  53. }
  55. #pragma mark 数组排序4-高级排序
  56. void arraySort4() {
  57. Student *stu1 = [Student studentWithFirstname:@”MingJie” lastname:@”Li” bookName:@”book1”];
  58. Student *stu2 = [Student studentWithFirstname:@”LongHu” lastname:@”Huang” bookName:@”book2”];
  59. Student *stu3 = [Student studentWithFirstname:@”LianJie” lastname:@”Li” bookName:@”book2”];
  60. Student *stu4 = [Student studentWithFirstname:@”Jian” lastname:@”Xiao” bookName:@”book1”];
  61. NSArray *array = [NSArray arrayWithObjects:stu1,stu2,stu3, stu4, nil nil];
  63. // 1.先按照书名进行排序
  64. // 这里的key写的是@property的名称
  65. NSSortDescriptor *bookNameDesc = [NSSortDescriptor sortDescriptorWithKey:@”book.name” ascending:YES];
  66. // 2.再按照姓进行排序
  67. NSSortDescriptor *lastnameDesc = [NSSortDescriptor sortDescriptorWithKey:@”lastname” ascending:YES];
  68. // 3.再按照名进行排序
  69. NSSortDescriptor *firstnameDesc = [NSSortDescriptor sortDescriptorWithKey:@”firstname” ascending:YES];
  70. // 按顺序添加排序描述器
  71. NSArray *descs = [NSArray arrayWithObjects:bookNameDesc, lastnameDesc, firstnameDesc, nil nil];
  73. NSArray *array2 = [array sortedArrayUsingDescriptors:descs];
  75. NSLog(@”array2:%@”, array2);
  76. }
